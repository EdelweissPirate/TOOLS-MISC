== Tech I ==

// creates several chunks and mpd to be used straight away (messy)
ffmpeg -i input.mp4 -vcodec copy -acodec copy output/output.mpd


== Tech II ==

// create audio stream
ffmpeg -i input.mp4 -vn -acodec libvorbis -ab 128k -dash 1 my_audio.webm

// create webm
ffmpeg -i input.mp4 -c:v libvpx-vp9 -keyint_min 150 -g 150 -tile-columns 4 -frame-parallel 1 -f webm -dash 1 -an -vf scale=640:360 -b:v 1000k -dash 1 video_640x360_1000k.webm


// create manifest (.mpd)
ffmpeg -f webm_dash_manifest -i output/video_640x360_1000k.webm -f webm_dash_manifest -i output/my_audio.webm -c copy -map 0 -map 1 -f webm_dash_manifest -adaptation_sets "id=0,streams=0 id=1,streams=1" output/my_video_manifest.mpd
